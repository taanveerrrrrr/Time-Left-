        <div id="buttons">
            <button onclick="incrementCount()">+1</button>
            <button onclick="resetCount()">Reset</button>
        </div>
    </div>

    <script>
        let count = 0;

        function incrementCount() {
            count++;
            updateCount();
        }

        function resetCount() {
            count = 0;
            updateCount();
        }

        function updateCount() {
            document.getElementById('count').innerText = count;
        }
    </script>
